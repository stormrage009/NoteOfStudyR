按id分组，

  • 如果corr中都是0 就”none”

  • 如果corr中都是1 就”both”

  • 如果corr中只有一个1，就输出1对应period

my_function <- function(corr, period){
  sum <- sum(corr)
  
  if (sum == 0){
    res <- "none"
  }
  
  if (sum == 2){
    res <- "both"
  }
  
  if (sum == 1){
    res <- period[corr == 1]
  }
  return(res)
}

data %>% 
  group_by(id) %>% 
  summarise(resp_period = my_function(corr, period))
## # A tibble: 4 x 2
##      id resp_period
##   <dbl> <chr>      
## 1     1 none       
## 2     2 b          
## 3     3 a          
## 4     4 both
